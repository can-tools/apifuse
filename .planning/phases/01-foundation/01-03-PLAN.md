  structlog.stdlib.add_log_level,

  structlog.stdlib.PositionalArgumentsFormatter(),
  drop_color_message_key,
  structlog.processors.TimeStamper(fmt="iso"),
  structlog.processors.StackInfoRenderer(),

structlog.configure() call:
  processors = shared_processors + [structlog.processors.format_exc_info, structlog.stdlib.ProcessorFormatter.wrap_for_formatter]
  logger_factory = structlog.stdlib.LoggerFactory()
  wrapper_class = structlog.make_filtering_bound_logger(log_level)
  cache_logger_on_first_use = True

ProcessorFormatter:
  foreign_pre_chain = shared_processors
  processors = [ProcessorFormatter.remove_processors_meta, renderer]

  handlers.clear() then addHandler(StreamHandler(sys.stdout) with ProcessorFormatter)
  setLevel(log_level)

uvicorn logger routing:
  "uvicorn" and "uvicorn.error": handlers.clear(), propagate = True
  "uvicorn.access": handlers.clear(), propagate = False

Renderer selection (D-01):
  is_dev=True → structlog.dev.ConsoleRenderer(colors=True)
  is_dev=False → structlog.processors.JSONRenderer()

Log level (D-02):
  is_dev=True → logging.DEBUG
